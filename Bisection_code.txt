#include<iostream>
#include<math.h>
#include<iomanip>
using namespace std;

int main(){
    int n = 20;
    float a[n] ,b[n];
    float fx[n];
    float x[n];
    a[0] = 2.00;
    b[0] = 3.00;
    x[0] = (a[0] + b[0])/2;
    for (int i = 0; i < n; i++)
    {
        fx[i] = pow(x[i], 3) - 2*x[i] - 5;
        if (fx[i] == 0)
        {
            cout<< x[i] << " is the real root of fx = 0" <<endl;
            n = i;
        }else if (fx[i] < 0)
        {
            x[i + 1] = (x[i] + b[i])/2;
            a[i + 1] = x[i];
            b[i + 1] = b[i];
        }else
        {
            x[i + 1] = (x[i] + a[i])/2;
            b[i + 1] = x[i];
            a[i + 1] = a[i];
        }
        
                
    }
    cout << left << setw(7) << "Sl no." 
         << left << setw(7) << "a" 
         << left << setw(7) << "b"
         << left << setw(7) << "x"
         << left << setw(7) << "fx" << endl;
    
    for (int i = 0; i < n; i++)
    {
        cout << left << setw(7) << i + 1 << " "
             << left << setw(7) << a[i] << " " 
             << left << setw(7) << b[i] << " "
             << left << setw(7) << x[i] << " "
             << left << setw(7) << fx[i] << endl;
    }
    


    


    return 0;
}
